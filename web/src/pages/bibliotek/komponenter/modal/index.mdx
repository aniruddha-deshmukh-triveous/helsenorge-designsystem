<MDXCustom.Preamble>
  En Modal informerer brukere om en oppgave og kan inneholde kritisk informasjon, kreve beslutninger eller involvere flere oppgaver. En
  modal har tre varianter normal, warning og error
</MDXCustom.Preamble>

## Usage
```javascript
import Modal from '@helsenorge/designsystem-react/components/Modal';

const ViewModal = ({open: boolean}) {
  const [showModal, setShowModal] = useState(open);

  return (
    <div>
      <button
        onClick={() => setShowModal(true)}
      >{props.buttonText}</button>
      {
        showModal && 
        <Modal
          title="Modal xx"
          onClose={() => setShowModal(false)}
        />
      }
    </div>
  );
});
```
